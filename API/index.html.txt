  <script>
    const messagesEl = document.getElementById('messages');
    const inputEl = document.getElementById('userInput');
    const errorEl = document.getElementById('error');
    const sendBtn = document.getElementById('sendBtn');
    const sendText = document.getElementById('sendText');
    const sendSpinner = document.getElementById('sendSpinner');

    function addMessage(text, type) {
      const wrapper = document.createElement('div');
      const label = document.createElement('div');
      const bubble = document.createElement('div');

      wrapper.style.display = 'flex';
      wrapper.style.flexDirection = 'column';
      wrapper.style.alignItems = type === 'me' ? 'flex-end' : 'flex-start';

      label.className = 'msg-label';
      label.textContent = type === 'me' ? 'נציג' : 'עוזר מכירות';

      bubble.className = 'msg ' + type;
      bubble.textContent = text;

      wrapper.appendChild(label);
      wrapper.appendChild(bubble);
      messagesEl.appendChild(wrapper);
      messagesEl.scrollTop = messagesEl.scrollHeight;
    }

    function setLoading(isLoading) {
      if (isLoading) {
        sendText.textContent = 'חושב...';
        sendSpinner.style.display = 'inline-block';
        sendBtn.disabled = true;
      } else {
        sendText.textContent = 'שלח';
        sendSpinner.style.display = 'none';
        sendBtn.disabled = false;
      }
    }

    async function sendToBot() {
      const txt = inputEl.value.trim();
      if (!txt) return;

      errorEl.textContent = '';
      addMessage(txt, 'me');
      inputEl.value = '';
      inputEl.focus();

      setLoading(true);

      try {
        const res = await fetch('/api/salesbot', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ text: txt })
        });

        if (!res.ok) {
          const data = await res.json().catch(() => ({}));
          throw new Error(data.error || ('שגיאה מהשרת: ' + res.status));
        }

        const data = await res.json();
        const reply = data.reply || 'לא התקבלה תשובה.';
        addMessage(reply, 'bot');
      } catch (err) {
        console.error(err);
        errorEl.textContent = err.message || 'אירעה שגיאה כללית.';
      } finally {
        setLoading(false);
      }
    }

    function quickFill(templateText) {
      inputEl.value = templateText;
      inputEl.focus();
    }

    inputEl.addEventListener('keydown', e => {
      if (e.key === 'Enter') sendToBot();
    });
  </script>
